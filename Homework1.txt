import boto3

# Retrieve the list of existing buckets
s3 = boto3.client('s3')

def main():
response = s3.list_buckets()
   print('Existing buckets:')
   for bucket in response['Buckets']:
       print(f'  {bucket["Name"]}')
       if(bucket.startswith("prod")){
           print (f'  {bucket["Name"]}')
       }
